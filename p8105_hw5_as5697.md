prop_city = function(x, y){
  
  test = prop.test(x,y) %>% 
    broom::tidy() %>% 
  select(estimate, conf.low, conf.high)
  
  test
  
}

prop_city<-map2(.x = homicide_group$n_unsolved, .y = homicide_group$n_total, ~prop_city(.x, .y)) %>%
